import { useState } from "react";
import { Link, useNavigate } from "react-router-dom";
// Hapus import apiService/formatLoginData karena kita langsung pakai fetch
// import { apiService, formatLoginData } from "../services/api.js";
import authManager from "../utils/auth.js";

export default function Login() {
  const [form, setForm] = useState({ email: "", password: "" });
  const [error, setError] = useState("");
  const [isLoading, setIsLoading] = useState(false);
  const navigate = useNavigate();

  const handleChange = (e) => {
    const { name, value } = e.target;
    setForm((prev) => ({
      ...prev,
      [name]: value,
    }));
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    setError("");
    setIsLoading(true);

    // Simple validation
    if (!form.email || !form.password) {
      setError("Please fill in all fields");
      setIsLoading(false);
      return;
    }

    try {
      // === LOKASI FETCH LOGIN ===
      // Di bawah ini kita langsung melakukan POST request ke backend:
      const res = await fetch("https://backend.serveruab.site/login", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          email: form.email,
          password: form.password,
        }),
      });

      if (!res.ok) {
        const msg = await res.text();
        throw new Error(msg || `Login failed (status ${res.status})`);
      }

      const responseData = await res.json();
      console.log("Login response received:", responseData);

      // Gunakan authManager untuk menyimpan token & data user
      const loginSuccess = authManager.login(responseData);

      if (loginSuccess) {
        console.log("Login successful! Auth state:", authManager.getAuthState());
        navigate("/");
        window.location.reload(); // Refresh navbar
      } else {
        setError("Login failed: No authentication token received from server");
        setIsLoading(false);
      }
    } catch (error) {
      console.error("Login error:", error);
      setError(
        error.message ||
          "Network error. Please check your connection and try again."
      );
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-ubud-dark-green to-ubud-light-green flex items-center justify-center p-4">
      <div className="bg-white/10 backdrop-blur-sm rounded-2xl shadow-2xl flex flex-col lg:flex-row w-full max-w-4xl overflow-hidden">
        {/* Left Side - Form */}
        <div className="flex-1 p-6 sm:p-8 flex flex-col justify-center">
          <div className="max-w-sm mx-auto w-full">
            <h2 className="text-2xl sm:text-3xl font-bold text-white mb-2">
              Welcome Back!
            </h2>
            <p className="text-ubud-cream mb-4 sm:mb-6 text-sm sm:text-base">
              Sign in to your account to continue
            </p>

            {error && (
              <div className="bg-red-500/20 border border-red-400 text-red-100 px-4 py-3 rounded-lg mb-4">
                {error}
              </div>
            )}

            <form onSubmit={handleSubmit} className="space-y-3 sm:space-y-4">
              <div>
                <input
                  type="email"
                  name="email"
                  value={form.email}
                  onChange={handleChange}
                  placeholder="Email address"
                  className="w-full px-3 sm:px-4 py-2.5 sm:py-3 rounded-lg bg-white/20 border border-white/30 text-white placeholder-white/70 focus:outline-none focus:ring-2 focus:ring-ubud-yellow focus:border-transparent text-sm sm:text-base"
                  required
                />
              </div>

              <div>
                <input
                  type="password"
                  name="password"
                  value={form.password}
                  onChange={handleChange}
                  placeholder="Password"
                  className="w-full px-3 sm:px-4 py-2.5 sm:py-3 rounded-lg bg-white/20 border border-white/30 text-white placeholder-white/70 focus:outline-none focus:ring-2 focus:ring-ubud-yellow focus:border-transparent text-sm sm:text-base"
                  required
                />
              </div>

              <div className="flex items-center justify-between text-ubud-cream text-sm">
                <label className="flex items-center cursor-pointer">
                  <input type="checkbox" className="mr-2 accent-ubud-yellow" />
                  Remember me
                </label>
                <a
                  href="#"
                  className="hover:text-ubud-yellow transition-colors"
                >
                  Forgot password?
                </a>
              </div>

              <button
                type="submit"
                disabled={isLoading}
                className="w-full bg-ubud-yellow text-ubud-dark-green font-bold py-3 rounded-lg hover:bg-yellow-400 transition-colors disabled:opacity-50 disabled:cursor-not-allowed"
              >
                {isLoading ? "Signing In..." : "Sign In"}

              <span className="text-ubud-cream">Don't have an account? </span>
              <Link
                to="/register"
                className="text-ubud-yellow font-bold hover:text-yellow-400 transition-colors"
              >
                Sign Up
              </Link>
            </div>
          </div>
        </div>

        {/* Right Side - Image */}
        <div className="hidden lg:block flex-1 relative">
          <div className="absolute inset-0 bg-gradient-to-br from-ubud-yellow/20 to-transparent"></div>
          <div
            className="h-full bg-cover bg-center"
            style={{

import { useState } from "react";
import { Link, useNavigate } from "react-router-dom";

export default function Register() {
  const [form, setForm] = useState({
    name: "",
    email: "",
    password: "",
    confirmPassword: "",
  });
  const [error, setError] = useState("");
  const [isLoading, setIsLoading] = useState(false);
  const navigate = useNavigate();

  const handleChange = (e) => {
    const { name, value } = e.target;
    setForm((prev) => ({ ...prev, [name]: value }));
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    setError("");
    setIsLoading(true);

    // Validasi dasar di sisi client
    if (!form.name || !form.email || !form.password || !form.confirmPassword) {
      setError("Please fill in all fields");
      setIsLoading(false);
      return;
    }
    if (form.password !== form.confirmPassword) {
      setError("Passwords do not match");
      setIsLoading(false);
      return;
    }
    if (form.password.length < 6) {
      setError("Password must be at least 6 characters");
      setIsLoading(false);
      return;
    }

    try {
      // === LOKASI FETCH REGISTER ===
      // Disesuaikan dengan skema PocketBase users (lihat screenshot kamu):
      const payload = {
        email: form.email,
        password: form.password,
        passwordConfirm: form.confirmPassword, // WAJIB di PB
        name: form.name,
        emailVisibility: true,                 // opsional, boleh true
      };

      const res = await fetch("https://backend.serveruab.site/register", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          Accept: "application/json",
        },
        body: JSON.stringify(payload),
      });

      // Baca response: coba parse JSON; kalau gagal, pakai text mentah
      const rawText = await res.text();
      let data = null;
      try { data = rawText ? JSON.parse(rawText) : null; }
      catch { /* ignore */ }

      if (!res.ok) {
        // Coba ambil detail error field-level gaya PocketBase
        const fieldErrors = data?.data
          ? Object.entries(data.data)
              .map(([field, detail]) => `${field}: ${detail?.message}`)
              .join(", ")
          : "";
        const msg = data?.message || rawText || `Failed to create record. (status ${res.status})`;
        throw new Error(fieldErrors ? `${msg} (${fieldErrors})` : msg);
      }

      alert("Registration successful! Please login.");
      navigate("/login");
    } catch (err) {
      console.error("Registration error:", err);
      setError(err.message || "Network error. Please try again.");
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-ubud-dark-green to-ubud-light-green flex items-center justify-center p-4">
      <div className="bg-white/10 backdrop-blur-sm rounded-2xl shadow-2xl flex w-full max-w-4xl overflow-hidden">
        {/* Left Side - Image */}
        <div className="hidden lg:block flex-1 relative">
          <div className="absolute inset-0 bg-gradient-to-br from-ubud-yellow/20 to-transparent"></div>
          <div
            className="h-full bg-cover bg-center"
            style={{
              backgroundImage:
                "url('https://images.unsplash.com/photo-1537953773345-d172ccf13cf1?ixlib=rb-4.0.3&ixid=M3wxMjA3fDB8MHxwaG90by1wYWdlfHx8fGVufDB8fHx8fA%3D%3D&auto=format&fit=crop&w=1000&q=80')",
            }}
          ></div>
          <div className="absolute inset-0 flex items-center justify-center">
            <div className="text-center text-white">
              <h3 className="text-2xl font-bold mb-2">Join Our Adventure</h3>
              <p className="text-ubud-cream">Create memories that last forever</p>
            </div>
          </div>
        </div>

        {/* Right Side - Form */}
        <div className="flex-1 p-8 flex flex-col justify-center">
          <div className="max-w-sm mx-auto w-full">
            <h2 className="text-3xl font-bold text-white mb-2">Create Account</h2>
            <p className="text-ubud-cream mb-6">Join us to enjoy the best adventure services</p>

            {error && (
              <div className="bg-red-500/20 border border-red-400 text-red-100 px-4 py-3 rounded-lg mb-4">
                {error}
              </div>
            )}

            <form onSubmit={handleSubmit} className="space-y-4">
              <div>
                <input
                  type="text"
                  name="name"
                  value={form.name}
                  onChange={handleChange}
                  placeholder="Full Name"
                  className="w-full px-4 py-3 rounded-lg bg-white/20 border border-white/30 text-white placeholder-white/70 focus:outline-none focus:ring-2 focus:ring-ubud-yellow focus:border-transparent"
                  required
                />
              </div>

              <div>
                <input
                  type="email"
                  name="email"
                  value={form.email}
                  onChange={handleChange}
                  placeholder="Email address"
                  className="w-full px-4 py-3 rounded-lg bg-white/20 border border-white/30 text-white placeholder-white/70 focus:outline-none focus:ring-2 focus:ring-ubud-yellow focus:border-transparent"
                  required
                />
              </div>

              <div>
                <input
                  type="password"
                  name="password"
                  value={form.password}
                  onChange={handleChange}
                  placeholder="Password"
                  className="w-full px-4 py-3 rounded-lg bg-white/20 border border-white/30 text-white placeholder-white/70 focus:outline-none focus:ring-2 focus:ring-ubud-yellow focus:border-transparent"
                  required
                />
              </div>

              <div>
                <input
                  type="password"
                  name="confirmPassword"
                  value={form.confirmPassword}
                  onChange={handleChange}
                  placeholder="Confirm Password"
                  className="w-full px-4 py-3 rounded-lg bg-white/20 border border-white/30 text-white placeholder-white/70 focus:outline-none focus:ring-2 focus:ring-ubud-yellow focus:border-transparent"
                  required

              </div>

              <button
                type="submit"
                disabled={isLoading}
                className="w-full bg-ubud-yellow text-ubud-dark-green font-bold py-3 rounded-lg hover:bg-yellow-400 transition-colors disabled:opacity-50 disabled:cursor-not-allowed"

                {isLoading ? "Creating Account..." : "Create Account"}
              </button>
            </form>

            <div className="mt-6 text-center">
              <span className="text-ubud-cream">Already have an account? </span>
              <Link
                to="/login"
                className="text-ubud-yellow font-bold hover:text-yellow-400 transition-colors"
              >
                Sign In
              </Link>
            </div>
          </div>
        </div>
      </div>
    </div>
  );

const toGallery = (item) => {
  if (!item?.img) return [images.jeep.group];

  // img bisa array atau string
  if (Array.isArray(item.img) && item.img.length) {
    return item.img.map((f) => buildFileUrl(item, f));
  }
  if (typeof item.img === "string" && item.img.trim().length) {
    return [buildFileUrl(item, item.img)];
  }
  return [images.jeep.group];
};

const getAuthToken = () => {
  try {
    const direct =
      localStorage.getItem("token") ||
      localStorage.getItem("access_token") ||
      localStorage.getItem("authToken") ||
      sessionStorage.getItem("token");
    if (direct) return direct;

    const pbAuth = JSON.parse(localStorage.getItem("pb_auth") || "null");
    if (pbAuth?.token) return pbAuth.token;

    const user = JSON.parse(localStorage.getItem("user") || "null");
    if (user?.token) return user.token;

    return null;
  } catch {
    return null;
  }
};

      const token = getAuthToken();

      // Kirim semua query param agar backend punya cukup konteks
      const payload = { order_id: orderId, ...allQueryParams };

      // === FETCH: Backend Midtrans Notification ===
      const res = await fetch(
        "https://backend.serveruab.site/payment/notification",
        {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            ...(token ? { Authorization: `Bearer ${token}` } : {}),
          },
          body: JSON.stringify(payload),
        }
      );

      const text = await res.text();
      let data = null;
      try {
        data = text ? JSON.parse(text) : null;
      } catch {
        data = { raw: text };
      }

      // Even if backend returns error, if we have transaction status from query params, use it
      const txStatus =
        (allQueryParams?.transaction_status ||
          data?.transaction_status ||
          data?.data?.transaction_status ||
          "settlement") + "";

      // If we have a valid transaction status, consider it successful notification
      if (txStatus && txStatus !== "undefined" && txStatus !== "null") {
        setNotifResult({
          message:
            data?.message || "Payment notification processed successfully",
          order_id:
            data?.order_id ||
            data?.data?.order_id ||
            allQueryParams?.order_id ||
            orderId,
          transaction_status: txStatus,
          status_code:
            data?.status_code ||
            data?.data?.status_code ||
            allQueryParams?.status_code ||
            "200",
        });

        // Status will be updated when we refresh history

        // If backend failed but we have valid status, don't throw error
        if (!res.ok) {
          console.warn("Backend notification failed but status updated locally:", {
            status: res.status,
            response: text,
            txStatus,
          });
        }
      } else if (!res.ok) {
        const msg =
          data?.message || data?.error || text || "Notification failed";
        throw new Error(msg);
      }
    } catch (err) {
      console.error("Notification error:", err);
      // Don't show error if we already have transaction status from URL params
      const txStatus = allQueryParams?.transaction_status;
      if (txStatus) {
        setNotifResult({
          message: "Payment status updated from transaction data",
          order_id: orderId,
          transaction_status: txStatus,
          status_code: "200",
        });
        // Status will be updated when we refresh history
      } else {
        setNotifError(err.message || "Failed to process payment notification");
      }
    } finally {
      setNotifLoading(false);
    }

import { useEffect, useRef, useState } from "react";

const ChatbotWidget = () => {
  const [isOpen, setIsOpen] = useState(false);
  const [isVisible, setIsVisible] = useState(false);
  const [messages, setMessages] = useState([
    {
      id: 1,
      text: "Hi 👋, how can I help you?",
      isBot: true,
      timestamp: new Date(),
    },
    {
      id: 2,
      text: "How to order the services?",
      isBot: true,
      timestamp: new Date(),
    },
  ]);
  const [inputMessage, setInputMessage] = useState("");
  const [isTyping, setIsTyping] = useState(false);
  const messagesEndRef = useRef(null);

  // Show widget when user scrolls down or after a delay
  useEffect(() => {
    const handleScroll = () => {
      const scrollTop =
        window.pageYOffset || document.documentElement.scrollTop;
      setIsVisible(scrollTop > 100); // Show after scrolling 100px (reduced from 200px)
    };

    // Show widget after 3 seconds if user hasn't scrolled
    const timer = setTimeout(() => {
      setIsVisible(true);
    }, 3000);

    window.addEventListener("scroll", handleScroll);

    return () => {
      window.removeEventListener("scroll", handleScroll);
      clearTimeout(timer);
    };
  }, []);

  // Auto scroll to bottom when new messages
  useEffect(() => {
    scrollToBottom();
  }, [messages]);

  const scrollToBottom = () => {
    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
  };

  const handleSendMessage = async () => {
    if (!inputMessage.trim()) return;

    const userMessage = {
      id: Date.now(),
      text: inputMessage,
      isBot: false,
      timestamp: new Date(),
    };

    setMessages((prev) => [...prev, userMessage]);
    const currentMessage = inputMessage;
    setInputMessage("");
    setIsTyping(true);

    try {
      // Call the backend API
      const response = await fetch("https://backend.serveruab.site/chat", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          message: currentMessage,
        }),
      });

      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }

      const data = await response.json();

      const botResponse = {
        id: Date.now() + 1,
        text: data.reply || "Sorry, I could not process your request.",
        isBot: true,
        timestamp: new Date(),
      };

      setMessages((prev) => [...prev, botResponse]);
    } catch (error) {
      console.error("Error calling chat API:", error);

      // Fallback to local response if API fails
      const botResponse = {
        id: Date.now() + 1,
        text: "Sorry, I am having trouble connecting to the server. Please try again later.",
        isBot: true,
        timestamp: new Date(),
      };

      setMessages((prev) => [...prev, botResponse]);
    } finally {
      setIsTyping(false);
    }
  };

  const handleKeyPress = (e) => {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      handleSendMessage();
    }
  };

  if (!isVisible) return null;

  return (
    <div className="fixed bottom-4 right-4 sm:bottom-6 sm:right-6 lg:bottom-8 lg:right-8 z-50">
      {/* Chat Widget */}
      {isOpen ? (
        <div className="bg-white rounded-2xl shadow-2xl border border-ubud-light-green w-[85vw] max-w-[320px] sm:w-[340px] md:w-[360px] lg:w-[380px] h-[60vh] max-h-[450px] sm:h-[420px] md:h-[460px] lg:h-[480px] flex flex-col animate-slideUp chatbot-mobile-optimized">
          {/* Header */}
          <div className="bg-ubud-dark-green text-white px-3 py-3 sm:px-4 sm:py-3 md:px-4 md:py-4 rounded-t-xl flex items-center justify-between">
            <div className="flex items-center space-x-2 sm:space-x-3 md:space-x-3">
              <div className="w-8 h-8 sm:w-9 sm:h-9 md:w-10 md:h-10 bg-white rounded-full flex items-center justify-center">
                <span className="text-base sm:text-lg md:text-xl">🤖</span>
              </div>
              <div>
                <h3 className="font-bold text-sm sm:text-base md:text-lg">Ubud Activity Bot</h3>
                <p className="text-xs sm:text-sm md:text-sm opacity-90">Your Travel Assistant</p>
              </div>
            </div>
            <div className="flex space-x-1">
              <button
                onClick={() => setIsOpen(false)}
                className="text-white hover:bg-white/20 p-1 rounded min-w-[28px] min-h-[28px] flex items-center justify-center chatbot-button-mobile"
              >
                <span className="text-base">-</span>
              </button>
              <button
                onClick={() => setIsOpen(false)}
                className="text-white hover:bg-white/20 p-1 rounded min-w-[28px] min-h-[28px] flex items-center justify-center chatbot-button-mobile"
              >
                <span className="text-base">x</span>
              </button>
            </div>
          </div>

          {/* Messages Area */}
          <div className="flex-1 px-3 py-2 sm:px-3 sm:py-3 md:px-4 md:py-3 overflow-y-auto bg-ubud-light-green/10 chatbot-scrollbar">
            <div className="space-y-2 sm:space-y-2 md:space-y-3">
              {messages.map((message) => (
                <div
                  key={message.id}
                  className={`flex ${
                    message.isBot ? "justify-start" : "justify-end"
                  }`}
                >
                  <div
                    className={`max-w-[80%] sm:max-w-[75%] md:max-w-[320px] px-3 sm:px-4 md:px-4 py-2 sm:py-3 md:py-3 rounded-xl sm:rounded-2xl chatbot-transition ${
                      message.isBot
                        ? "bg-white text-ubud-dark-green shadow-sm border border-ubud-light-green/20"
                        : "bg-ubud-dark-green text-white shadow-sm"
                    }`}
                  >
                    <p className="text-sm sm:text-sm md:text-base leading-relaxed chatbot-text-mobile">{message.text}</p>
                  </div>
                </div>
              ))}

              {isTyping && (
                <div className="flex justify-start">
                  <div className="bg-white text-ubud-dark-green px-3 sm:px-4 py-2 sm:py-3 rounded-lg sm:rounded-2xl shadow-sm">
                    <div className="flex space-x-1 sm:space-x-1">
                      <div className="w-1.5 h-1.5 sm:w-2 sm:h-2 bg-ubud-dark-green rounded-full animate-bounce"></div>
                      <div
                        className="w-1.5 h-1.5 sm:w-2 sm:h-2 bg-ubud-dark-green rounded-full animate-bounce"
                        style={{ animationDelay: "0.1s" }}
                      ></div>
                      <div
                        className="w-1.5 h-1.5 sm:w-2 sm:h-2 bg-ubud-dark-green rounded-full animate-bounce"
                        style={{ animationDelay: "0.2s" }}
                      ></div>
                    </div>
                  </div>
                </div>
              )}
              <div ref={messagesEndRef} />
            </div>
          </div>

          {/* Input Area */}
          <div className="px-3 py-2 sm:px-3 sm:py-3 md:px-4 md:py-3 border-t border-ubud-light-green/20 bg-white rounded-b-xl">
            <div className="flex space-x-2 sm:space-x-2 md:space-x-3">
              <input
                type="text"
                value={inputMessage}
                onChange={(e) => setInputMessage(e.target.value)}
                onKeyDown={handleKeyPress}
                placeholder="Type your message..."
                className="flex-1 px-3 sm:px-4 md:px-4 py-2 sm:py-3 md:py-3 border border-ubud-light-green/30 rounded-full focus:outline-none focus:border-ubud-dark-green focus:ring-2 focus:ring-ubud-light-green/20 text-sm sm:text-base md:text-base min-h-[44px] sm:min-h-[48px] md:min-h-[48px] chatbot-transition"
              />
              <button
                onClick={handleSendMessage}
                disabled={!inputMessage.trim()}
                className="bg-ubud-dark-green text-white rounded-full hover:bg-ubud-light-green disabled:opacity-50 disabled:cursor-not-allowed chatbot-transition flex items-center justify-center chatbot-button-mobile group"
                style={{
                    minWidth: 'clamp(44px, 10vw, 48px)',
                    minHeight: 'clamp(44px, 10vw, 48px)',
                    padding: 'clamp(8px, 2vw, 12px)'
                }}
              >
                <svg
                  viewBox="0 0 24 24"
                  fill="none"
                  xmlns="http://www.w3.org/2000/svg"
                  className="transform rotate-45 transition-transform group-hover:scale-110"
                  style={{
                    width: 'clamp(14px, 3vw, 18px)',
                    height: 'clamp(14px, 3vw, 18px)'
                  }}
                >
                  <path
                    d="M7 17L17 7M17 7H7M17 7V17"
                    stroke="currentColor"
                    strokeWidth="2"
                    strokeLinecap="round"
                    strokeLinejoin="round"
                  />
                </svg>
              </button>
            </div>
          </div>
        </div>
      ) : (
        /* Floating Button */
        <button
          onClick={() => setIsOpen(true)}
          className="bg-ubud-dark-green text-white rounded-full shadow-lg hover:bg-ubud-light-green transition-all duration-300 hover:scale-105 group relative overflow-hidden chatbot-floating-button"
          style={{
            width: 'clamp(56px, 12vw, 64px)',
            height: 'clamp(56px, 12vw, 64px)',
            padding: 'clamp(12px, 2.5vw, 16px)'
          }}
        >
          <div className="flex items-center justify-center w-full h-full">
            <span className="text-xl sm:text-2xl transition-transform group-hover:scale-110">💭</span>
          </div>
          {/* Pulse animation ring */}
          <div className="absolute inset-0 rounded-full bg-ubud-dark-green opacity-30 animate-ping"></div>
        </button>
      )}
    </div>
  );
};

export default ChatbotWidget;
